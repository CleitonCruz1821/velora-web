# Interpretador simples da linguagem Velora
# Feito por ChatGPT + Você 🚀

def interpretar_linha(linha):
    linha = linha.strip()

    if linha.startswith("escreva "):
        conteudo = linha[len("escreva "):].strip()
        if conteudo.startswith('"') and conteudo.endswith('"'):
            print(conteudo[1:-1])
        else:
            try:
                print(eval(conteudo))
            except Exception as e:
                print(f"[Erro de Avaliação]: {e}")

    elif linha.startswith("se "):
        condicao = linha[len("se "):].strip()
        try:
            if eval(condicao):
                return "executar"
            else:
                return "pular"
        except Exception as e:
            print(f"[Erro na condição]: {e}")
            return "erro"

    elif linha.startswith("senão"):
        return "senão"

    else:
        try:
            exec(linha)
        except Exception as e:
            print(f"[Erro de Execução]: {e}")

def interpretar_velora(codigo):
    linhas = codigo.split('\n')
    modo = "executar"

    for linha in linhas:
        linha = linha.strip()
        if not linha or linha.startswith("#"):  # Ignorar linhas vazias ou comentários
            continue

        if linha.startswith("se "):
            modo = interpretar_linha(linha)
        elif linha.startswith("senão"):
            modo = "inverter"
        else:
            if modo == "executar":
                interpretar_linha(linha)
            elif modo == "inverter":
                interpretar_linha(linha)
                modo = "pular"
            elif modo == "pular":
                continue

function executarVelora() {
const codigo = document.getElementById("codigoVelora").value;
const linhas = codigo.split('\n');
const saida = document.getElementById("saidaVelora");
saida.textContent = ""; // Limpa a saída anterior

let modo = "executar";

linhas.forEach(linha => {
linha = linha.trim();
if (!linha || linha.startsWith("#")) {
return; // Ignora linhas vazias ou comentários
}

if (linha.startsWith("escreva ")) {
let conteudo = linha.substring(8).trim();
if (conteudo.startsWith('"') && conteudo.endsWith('"')) {
saida.textContent += conteudo.slice(1, -1) + "\n";
} else {
try {
saida.textContent += eval(conteudo) + "\n";
} catch (e) {
saida.textContent += "[Erro de avaliação]: " + e + "\n";
}
}
} else if (linha.startsWith("se ")) {
let condicao = linha.substring(3).trim();
try {
modo = eval(condicao) ? "executar" : "pular";
} catch (e) {
saida.textContent += "[Erro na condição]: " + e + "\n";
modo = "erro";
}
} else if (linha.startsWith("senão")) {
modo = modo === "pular" ? "executar" : "pular";
} else {
if (modo === "executar") {
try {
eval(linha);
} catch (e) {
saida.textContent += "[Erro de execução]: " + e + "\n";
}
}
}
});
}
</script>
